heresiarch_begin_converting_county_effect = {
    $HERESIARCH$ = { save_scope_as = heresiarch }
    set_variable = {
        # when currently_converting reaches 100 it will flip the county to the heresiarch's faith
        name = converting_county
        value = 0
    }
}

heresiarch_county_conversion_tick_effect = {
    if = {
        limit = {
            exists = $HERESIARCH$.var:converting_county
        }
        change_variable = {
            name = converting_county
            add = $HERESIARCH$.heresiarch_county_conversion_uptick_value
        }
    }
    if = {
        limit = {
            $HERESIARCH$.var:converting_county >= 100
        }
        $HERESIARCH$.location.county = {
            set_county_faith = $HERESIARCH$.faith
            trigger_event = organic_heresy.1000
        }
    }
}
