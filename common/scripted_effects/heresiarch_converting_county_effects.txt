heresiarch_begin_converting_county_effect = {
    $HERESIARCH$ = { save_scope_as = heresiarch }
    set_variable = {
        # when currently_converting reaches 100 it will flip the county to the heresiarch's faith
        name = converting_county
        value = 0
    }
}

heresiarch_county_conversion_tick_effect = {
    $HERESIARCH$ = { save_scope_as = heresiarch }
    if = {
        limit = {
            exists = scope:heresiarch.var:converting_county
        }
        change_variable = {
            name = converting_county
            add = scope:heresiarch.heresiarch_county_conversion_uptick_value
        }
    }
    if = {
        limit = {
            scope:heresiarch.var:converting_county >= 100
        }
        scope:heresiarch = {
            trigger_event = heresiarch.3000
        }
    }
}
