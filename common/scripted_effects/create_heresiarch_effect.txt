create_heresiarch_effect = {
    $STRONGHOLD_COUNTY$ = {
        save_scope_as = stronghold
    }
    scope:stronghold = {
        create_character = {
            location = scope:stronghold
            template = heresiarch_template
            save_scope_as = new_heresiarch
        }
    }
    scope:new_heresiarch = {
        set_variable = {
            name = heresiarch_stronghold
            value = scope:stronghold
        }
    }
}
