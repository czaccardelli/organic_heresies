# called in addition to base on_faith_monthly actions in common/on_action/religion_on_action
on_faith_monthly = {
	on_actions = {
		faith_heresy_spread_pulse
	}	
}

faith_heresy_spread_pulse = {
    trigger = {
        any_province = {
            county = {
                has_county_modifier = heretic_stronghold_modifier
                faith = root
            }
        }
    }
    random_events = {
        chance_to_happen = 50 #3 # ~31% chance to happen at least once per year
        chance_of_no_event = {
            value = 100
            # low fervor in the heresy will lower chance of heresy spread from heretic strongholds
            subtract = fervor
        }
        100 = organic_heresy.0011
    }
        
}
